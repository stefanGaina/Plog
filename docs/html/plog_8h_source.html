272 /** ***************************************************************************************************
273  * @brief Enumerates the bits representing if the log severity level is enabled.
274  *****************************************************************************************************/
275 typedef enum e_plog_SeverityLevel_t
276 {
277  E_PLOG_SEVERITY_LEVEL_FATAL = (1 << 0), /**< If bit is set fatal logs are enabled. */
278  E_PLOG_SEVERITY_LEVEL_ERROR = (1 << 1), /**< If bit is set error logs are enabled. */
279  E_PLOG_SEVERITY_LEVEL_WARN = (1 << 2), /**< If bit is set warn logs are enabled. */
280  E_PLOG_SEVERITY_LEVEL_INFO = (1 << 3), /**< If bit is set info logs are enabled. */
281  E_PLOG_SEVERITY_LEVEL_DEBUG = (1 << 4), /**< If bit is set debug logs are enabled. */
282  E_PLOG_SEVERITY_LEVEL_TRACE = (1 << 5), /**< If bit is set trace logs are enabled. */
283  E_PLOG_SEVERITY_LEVEL_VERBOSE = (1 << 6) /**< If bit is set verbose logs are enabled. */
284 } plog_SeverityLevel_t;
